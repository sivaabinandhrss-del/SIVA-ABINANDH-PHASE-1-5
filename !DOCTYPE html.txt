  <script>
    function navigate(page) {
      const content = document.getElementById("content");
      const links = document.querySelectorAll("nav a");

      // Remove active class from all links
      links.forEach(link => link.classList.remove("active"));

      // Add active class to clicked link
      document.getElementById(page + "Link").classList.add("active");

      // Change content based on page
      if (page === "home") {
        content.innerHTML = `
          <h1>🏠 Welcome Home</h1>
          <p>This is the home section of your single page app.</p>
        `;
      } 
      else if (page === "about") {
        content.innerHTML = `
          <h1>ℹ️ About Us</h1>
          <p>We are a small team learning how to build simple SPAs using HTML, CSS, and JS.</p>
        `;
      } 
      else if (page === "services") {
        content.innerHTML = `
          <h1>🛠 Our Services</h1>
          <ul style="list-style:none; padding:0;">
            <li>✔️ Web Design</li>
            <li>✔️ Frontend Development</li>
            <li>✔️ UI/UX Design</li>
          </ul>
        `;
      } 
      else if (page === "contact") {
        content.innerHTML = `
          <h1>📞 Contact Us</h1>
          <p>Email: support@example.com</p>
          <p>Phone: +91 98765 43210</p>
        `;
      }
    }

    // Load home page by default
    navigate("home");
